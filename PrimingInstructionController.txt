package PrimingInstructionApp;

import javafx.event.ActionEvent;
import javafx.fxml.FXML;
import javafx.fxml.FXMLLoader;
import javafx.fxml.Initializable;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.control.Label;
import javafx.scene.control.PasswordField;
import javafx.scene.control.TextField;
import javafx.scene.layout.Pane;
import javafx.stage.Stage;
import javafx.util.Duration;

import java.io.IOException;
import java.net.URL;
import java.util.ResourceBundle;

public class PrimingInstructionController implements Initializable {

    PrimingInstructionModel PrimingInstructionModel = new PrimingInstructionModel();

    @FXML private Label dbStatusId;
    @FXML private Button signInButtonId;

    @Override

    public void initialize(URL url, ResourceBundle rb) {
        if(this.PrimingInstructionModel.isDatabaseConnected()){
            this.dbStatusId.setText("Database connected");
        }else{
            this.dbStatusId.setText("Database NOT connected");
        }
    }

    @FXML

    public void showPrimingInstruction(){
        try{
            Stage PrimingInstructionStage = new Stage();
            FXMLLoader PrimingInstructionLoader = new FXMLLoader();
            Pane PrimingInstructionRoot = (Pane)PrimingInstructionLoader.load(getClass().getResource("/Instructions/PrimingInstruction.fxml").openStream());

            AdminController PrimingInstructionController = (AdminController)PrimingInstructionLoader.getController();
            Scene scene = new Scene(PrimingInstructionRoot);
            PrimingInstructionStage.setScene(scene);
            PrimingInstructionStage.setTitle("PrimingInstruction Point");
            PrimingInstructionStage.setResizable(false);
            PrimingInstructionStage.show();

        }catch (IOException ex){
            ex.printStackTrace();
        }

}

}
